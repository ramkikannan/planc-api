135  void in_order_reuse_MTTKRP(long int n, double *out, bool colmajor,
136  double &multittv_time, double &mttkrp_time) {
137  // ktensor *Y = m_local_Y;
138  direction D;
139  multittv_time = 0;
140  mttkrp_time = 0;
141 
142  if (n == 0) {
146  D = ::direction::right; // Contracting over the right modes
147  LR_Ktensor_Reordering_existingY(
148  m_local_Y, projection_Ktensor, s + 1,
149  opposite_direction(D)); // s+1 because s is included in the left
150 
151  if (projection_Ktensor->nmodes == 1) { // factor matrix output PM
159  tic();
160  partial_MTTKRP_with_KRP_output_FM(D, m_local_Y, m_local_T, num_threads);
161  mttkrp_time += toc();
162 
163  } else { // tensor output PM
173  tic();
174  partial_MTTKRP_with_KRP_output_T(s, D, m_local_Y, m_local_T,
175  projection_Tensor, num_threads);
176  mttkrp_time += toc();
177  tic();
178  multi_TTV_with_KRP_output_FM(D, projection_Tensor, projection_Ktensor,
179  num_threads);
180  multittv_time += toc();
181  }
182  } else if (n == s + 1) {
186  D = ::direction::left; // Contracting over the left modes
187  LR_Ktensor_Reordering_existingY(
188  m_local_Y, projection_Ktensor, s,
189  opposite_direction(D)); // s because s is not included on the right
190  if (projection_Ktensor->nmodes == 1) { // factor matrix output PM
191  tic();
192  partial_MTTKRP_with_KRP_output_FM(D, m_local_Y, m_local_T, num_threads);
193  mttkrp_time += toc();
194  } else {
195  tic();
196  partial_MTTKRP_with_KRP_output_T(s, D, m_local_Y, m_local_T,
197  projection_Tensor, num_threads);
198  mttkrp_time += toc();
199 
200  // op(D) because multi ttvs in this tree structure are always right
201  tic();
202  multi_TTV_with_KRP_output_FM(opposite_direction(D), projection_Tensor,
203  projection_Ktensor, num_threads);
204  multittv_time += toc();
205  }
206  } else {
207  D = ::direction::left;
208  if (projection_Ktensor->nmodes == 2) {
209  // A single left contraction to output a factor martix
210  tic();
211  multi_TTV_with_KRP_output_FM(D, projection_Tensor, projection_Ktensor,
212  num_threads);
213  multittv_time += toc();
214  } else {
219  tic();
220  multi_TTV_with_KRP_output_T(0, D, projection_Tensor, projection_Ktensor,
221  buffer_Tensor, num_threads);
222  multittv_time += toc();
223 
224  LR_tensor_Reduction(buffer_Tensor, projection_Tensor,
225  buffer_Tensor->nmodes, D);
226  tensor_data_swap(projection_Tensor, buffer_Tensor);
227 
228  remove_mode_Ktensor(projection_Ktensor,
229  0); // remove the 0th factor matrix and mode from
230  // the projection_Ktensor
231  tic();
232  multi_TTV_with_KRP_output_FM(::direction::right, projection_Tensor,
233  projection_Ktensor, num_threads);
234  multittv_time += toc();
235  }
236  }
237  if (colmajor) {
238  TransposeM(m_local_Y->factors[n], out, m_local_Y->rank,
239  m_local_Y->dims[n]);
240  } else {
241  std::memcpy(out, m_local_Y->factors[n],
242  sizeof(double) * m_local_Y->rank * m_local_Y->dims[n]);
243  }
244  }
