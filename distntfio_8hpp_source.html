45  void randomLowRank(const UVEC i_global_dims, const UVEC i_local_dims,
46  const UVEC i_start_rows, const UWORD i_k) {
47  // start with the same seed_idx with the global dimensions
48  // on all the MPI processor.
49  NCPFactors global_factors(i_global_dims, i_k, false);
50  global_factors.randu(kW_seed_idx);
51  global_factors.normalize();
52  int tensor_modes = global_factors.modes();
53  NCPFactors local_factors(i_local_dims, i_k, false);
54  UWORD start_row, end_row;
55  for (int i = 0; i < local_factors.modes(); i++) {
56  start_row = i_start_rows(i);
57  end_row = start_row + i_local_dims(i) - 1;
58  local_factors.factor(i) =
59  global_factors.factor(i).rows(start_row, end_row);
60  }
61  local_factors.rankk_tensor(m_A);
62  }

64  void build_local_tensor() {
65  // assumption is that the m_A has global tensor with it now.
66  // Local_size = global_size/proc_grids // mode wise division. Local_size is
67  // a vector of size modes For i over modes // at the end of this loop we
68  // will have the start_idx and end_idx of every mode for every rank
69  // Start_idx(i) = MPI_FIBER_RANK(i) * local_dims(i)
70  // End_idx(i) = start_idx(i) + local_size(i) - 1
71  // List of subscripts = {set of subs for every mode} .
72  // Find the Cartesian product of the list of idxs.
73  // This will give us the list of tensor global subscripts
74  // and should be in the row - major order.
75  // Call the sub2ind of the tensor for all the global
76  // subscripts and extract those data from the global data.
77 
78  // Local_size = global_size/proc_grids
79  UVEC global_dims = m_A.dimensions();
80  UVEC local_dims = global_dims / this->m_mpicomm.proc_grids();
81  size_t num_modes = m_A.modes();
82  std::vector<std::vector<size_t>> idxs_for_mode;
83  size_t current_start_idx, current_end_idx, num_idxs_for_mode;
84  // For i over modes
85  for (size_t i = 0; i < num_modes; i++) {
86  // Start_idx(i) = MPI_FIBER_RANK(i) * global_dims(i)
87  current_start_idx = MPI_FIBER_RANK(i) * local_dims(i);
88  // current_end_idx = current_start_idx + local_dims(i);
89  // num_idxs_for_mode = current_end_idx - current_start_idx;
90  // UVEC idxs_current_mode = arma::linspace<UVEC>(current_start_idx,
91  // current_end_idx - 1,
92  // num_idxs_for_mode);
93  std::vector<size_t> idxs_current_mode(local_dims(i));
94  std::iota(std::begin(idxs_current_mode), std::end(idxs_current_mode),
95  current_start_idx);
96  // arma::conv_to<std::vector<size_t>>::from(idxs_current_mode);
97  idxs_for_mode.push_back(idxs_current_mode);
98  }
99  // we have to now determing the cartesian product of this set.
100  std::vector<std::vector<size_t>> global_idxs =
101  cartesian_product(idxs_for_mode);
102  Tensor local_tensor(local_dims);
103  UVEC global_idxs_uvec = arma::zeros<UVEC>(global_idxs[0].size());
104  for (size_t i = 0; i < global_idxs.size(); i++) {
105  global_idxs_uvec = arma::conv_to<UVEC>::from(global_idxs[i]);
106  local_tensor.m_data[i] = m_A.at(global_idxs_uvec);
107  }
108  this->m_A = local_tensor;
109  }

157  UVEC read_dist_tensor(const std::string filename,
158  UVEC *start_idxs_uvec = NULL) {
159  // all processes reading the file_name.info file.
160  std::string filename_no_extension =
161  filename.substr(0, filename.find_last_of("."));
162  filename_no_extension.append(".info");
163  std::ifstream ifs;
164  int modes;
165  PRINTROOT("Reading tensor" << filename);
166  // info file always in text mode
167  ifs.open(filename_no_extension, std::ios_base::in);
168  // write modes
169  ifs >> modes;
170  // dimension of modes
171  // UVEC global_dims = arma::zeros<UVEC>(this->m_modes);
172  int *global_dims = new int[modes];
173  int *local_dims = new int[modes];
174  int *start_idxs = new int[modes];
175  this->m_local_dims = arma::zeros<UVEC>(modes);
176  this->m_global_dims = arma::zeros<UVEC>(modes);
177  UVEC tmp_start_idxs_uvec = arma::zeros<UVEC>(modes);
178  UVEC tmp_proc_grids = this->m_mpicomm.proc_grids();
179  for (int i = 0; i < modes; i++) {
180  ifs >> global_dims[i];
181  this->m_global_dims[i] = global_dims[i];
182  local_dims[i] = itersplit(global_dims[i], tmp_proc_grids[i],
183  this->m_mpicomm.fiber_rank(i));
184  this->m_local_dims[i] = local_dims[i];
185  start_idxs[i] = startidx(global_dims[i], tmp_proc_grids[i],
186  this->m_mpicomm.fiber_rank(i));
187  if (start_idxs_uvec != NULL) start_idxs_uvec[i] = start_idxs[i];
188  tmp_start_idxs_uvec[i] = start_idxs[i];
189  }
190  ifs.close();
191  DISTPRINTINFO("global dims::" << this->m_global_dims
192  << "Local Tensor Dims::" << this->m_local_dims
193  << "::start_idxs::" << tmp_start_idxs_uvec);
194  Tensor rc(this->m_local_dims);
195  // Create the datatype associated with this layout
196  MPI_Datatype view;
197  MPI_Type_create_subarray(modes, global_dims, local_dims, start_idxs,
198  MPI_ORDER_FORTRAN, MPI_DOUBLE, &view);
199  MPI_Type_commit(&view);
200  // Open the file
201  MPI_File fh;
202  // const MPI_Comm &comm = Y->getDistribution()->getComm(true);
203  // get confirmed with grey
204  int ret = MPI_File_open(MPI_COMM_WORLD, filename.c_str(), MPI_MODE_RDONLY,
205  MPI_INFO_NULL, &fh);
206  if (ret != MPI_SUCCESS) {
207  DISTPRINTINFO("Error: Could not read file " << filename << std::endl);
208  }
209  // Set the view
210  MPI_Offset disp = 0;
211  MPI_File_set_view(fh, disp, MPI_DOUBLE, view, "native", MPI_INFO_NULL);
212  // Read the file
213  int count = rc.numel();
214  DISTPRINTINFO("reading::" << count << "::in gbs::"
215  << (count * 8.0) / (1024 * 1024 * 1024));
216  assert(count * 8 <= std::numeric_limits<int>::max());
217  if (ISROOT && 8 * count > std::numeric_limits<int>::max()) {
218  PRINTROOT("file read size ::" << 8.0 * count << " > 2GB" << std::endl);
219  }
220  MPI_Status status;
221  ret = MPI_File_read_all(fh, rc.m_data, count, MPI_DOUBLE, &status);
222  int nread;
223  MPI_Get_count(&status, MPI_DOUBLE, &nread);
224  if (ret != MPI_SUCCESS) {
225  DISTPRINTINFO("Error: Could not read file " << filename << std::endl);
226  }
227  // Close the file
228  MPI_File_close(&fh);
229  // Free the datatype
230  MPI_Type_free(&view);
231 
232  // free the allocated things.
233  delete[] global_dims;
234  delete[] local_dims;
235  delete[] start_idxs;
236  this->m_A = rc;
237  rc.clear();
238  return this->m_global_dims;
239  }
240  /*
241  Reading from real input file.
242  Expecting a .tensor text file and .bin file.
243  .info text file will be written by root processor alone
244  .bin file will be an mpi io file
245  Read distributed tensor
246 */
247  void write_dist_tensor(const std::string filename,
248  const Tensor &local_tensor) {
249  // all processes reading the file_name.info file.
250  std::string filename_no_extension =
251  filename.substr(0, filename.find_last_of("."));
252  int modes = local_tensor.modes();
253  int *gsizes = new int[modes];
254  int *lsizes = new int[modes];
255  int *starts = new int[modes];
256  UVEC tmp_proc_grids = this->m_mpicomm.proc_grids();
257  for (int i = 0; i < modes; i++) {
258  lsizes[i] = local_tensor.dimension(i);
259  MPI_Allreduce(&lsizes[i], &gsizes[i], 1, MPI_INT, MPI_SUM,
260  this->m_mpicomm.fiber(i));
261  starts[i] =
262  startidx(gsizes[i], tmp_proc_grids[i], this->m_mpicomm.fiber_rank(i));
263  }
264  // info file always in text mode
265  if (ISROOT) {
266  filename_no_extension.append(".info");
267  std::ofstream ofs;
268  ofs.open(filename_no_extension, std::ios_base::out | std::ios::app);
269  // write modes
270  ofs << modes << std::endl;
271  // dimension of modes
272  for (int i = 0; i < modes; i++) {
273  ofs << gsizes[i] << std::endl;
274  }
275  ofs.close();
276  }
277  MPI_Barrier(MPI_COMM_WORLD);
278  // Now each of write to the file.
279  // Create the datatype associated with this layout
280 
281  MPI_Datatype view;
282  MPI_Type_create_subarray(modes, gsizes, lsizes, starts, MPI_ORDER_FORTRAN,
283  MPI_DOUBLE, &view);
284  MPI_Type_commit(&view);
285 
286  // Open the file
287  MPI_File fh;
288  int ret =
289  MPI_File_open(MPI_COMM_WORLD, filename.c_str(),
290  MPI_MODE_CREATE | MPI_MODE_WRONLY, MPI_INFO_NULL, &fh);
291  if (ISROOT && ret != MPI_SUCCESS) {
292  DISTPRINTINFO("Error: Could not open file " << filename << std::endl);
293  }
294 
295  // Set the view
296  MPI_Offset disp = 0;
297  MPI_File_set_view(fh, disp, MPI_DOUBLE, view, "native", MPI_INFO_NULL);
298 
299  // Write the file
300  int count = local_tensor.numel();
301  assert(count <= std::numeric_limits<int>::max());
302  MPI_Status status;
303  ret =
304  MPI_File_write_all(fh, local_tensor.m_data, count, MPI_DOUBLE, &status);
305  if (ret != MPI_SUCCESS) {
306  DISTPRINTINFO("Error: Could not write file " << filename << std::endl);
307  }
308  // Close the file
309  MPI_File_close(&fh);
310  // Free the datatype
311  MPI_Type_free(&view);
312  // free the allocated memories
313  delete[] gsizes;
314  delete[] lsizes;
315  delete[] starts;
316  }

324  void readInput(const std::string file_name, UVEC i_global_dims,
325  UVEC i_proc_grids, UWORD k = 0, double sparsity = 0) {
326  // INFO << "readInput::" << file_name << "::" << distio << "::"
327  // << m << "::" << n << "::" << pr << "::" << pc
328  // << "::" << this->MPI_RANK << "::" << this->m_mpicomm.size() <<
329  // std::endl;
330  this->m_global_dims = i_global_dims;
331  std::string rand_prefix("rand_");
332  if (!file_name.compare(0, rand_prefix.size(), rand_prefix)) {
333  this->m_local_dims = arma::zeros<UVEC>(i_proc_grids.n_rows);
334  UVEC start_rows = arma::zeros<UVEC>(i_proc_grids.n_rows);
335  // Calculate tensor local dimensions
336  for (int mode = 0; mode < this->m_local_dims.n_rows; mode++) {
337  int slice_num = this->m_mpicomm.slice_num(mode);
338  this->m_local_dims[mode] =
339  itersplit(i_global_dims[mode], i_proc_grids[mode], slice_num);
340  start_rows[mode] =
341  startidx(i_global_dims[mode], i_proc_grids[mode], slice_num);
342  }
343  if (!file_name.compare("rand_uniform")) {
344  // Tensor temp(i_global_dims / i_proc_grids);
345  Tensor temp(this->m_local_dims, start_rows);
346  this->m_A = temp;
347  // generate again. otherwise all processes will have
348  // same input tensor because of the same seed.
349  this->m_A.randu(449 * MPI_RANK + 677);
350  } else if (!file_name.compare("rand_lowrank")) {
351  randomLowRank(i_global_dims, this->m_local_dims, start_rows, k);
352  this->m_A.set_idx(start_rows);
353  }
354  } else {
355  read_dist_tensor(file_name);
356  }
