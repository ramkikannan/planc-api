46  * A random matrix is always needed for sparse case
47  * to get the pattern. That is., the indices where
48  * numbers are being filled. We will use this pattern

51  void randMatrix(const std::string type, const int primeseedidx,
52  const double sparsity, MATTYPE* X) {
53  if (primeseedidx == -1) {
54  arma::arma_rng::set_seed_random();
55  } else {
56  arma::arma_rng::set_seed(random_sieve(primeseedidx));
57  }
58 #ifdef DEBUG_VERBOSE
59  DISTPRINTINFO("randMatrix::" << primeseedidx << "::sp=" << sparsity);
60 #endif
61 #ifdef BUILD_SPARSE
62  if (type == "uniform" || type == "lowrank") {
63  (*X).sprandu((*X).n_rows, (*X).n_cols, sparsity);
64  } else if (type == "normal") {
65  (*X).sprandn((*X).n_rows, (*X).n_cols, sparsity);
66  }
67  SP_MAT::iterator start_it = (*X).begin();
68  SP_MAT::iterator end_it = (*X).end();
69  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
70  double currentValue = (*it);
71  (*it) = ceil(kalpha * currentValue + kbeta);
72  if ((*it) < 0) (*it) = kbeta;
73  }
74  // for (uint i=0; i<(*X).n_nonzero;i++){
75  // double currentValue = (*X).values[i];
76  // currentValue *= kalpha;
77  // currentValue += kbeta;
78  // currentValue = ceil(currentValue);
79  // if (currentValue <= 0) currentValue=kbeta;
80  // (*X).values[i]=currentValue;
81  // }
82 #else
83  if (type == "uniform") {
84  (*X).randu();
85  } else if (type == "normal") {
86  (*X).randn();
87  }
88  (*X) = kalpha * (*X) + kbeta;
89  (*X) = ceil(*X);
90  (*X).elem(find((*X) < 0)).zeros();
91 #endif
92  }

95  /* normalization */
96  void normalize(normtype i_normtype) {
97  ROWVEC globalnormA = arma::zeros<ROWVEC>(m_A.n_cols);
98  ROWVEC normc = arma::zeros<ROWVEC>(m_A.n_cols);
99  MATTYPE normmat = arma::zeros<MATTYPE>(m_A.n_rows, m_A.n_cols);
100  switch (m_distio) {
101  case ONED_ROW:
102  if (i_normtype == L2NORM) {
103  normc = arma::sum(arma::square(m_A));
104  MPI_Allreduce(normc.memptr(), globalnormA.memptr(), m_A.n_cols,
105  MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
106 
107  } else if (i_normtype == MAXNORM) {
108  normc = arma::max(m_A);
109  MPI_Allreduce(normc.memptr(), globalnormA.memptr(), m_A.n_cols,
110  MPI_DOUBLE, MPI_MAX, MPI_COMM_WORLD);
111  }
112 
113  break;
114  case ONED_COL:
115  case ONED_DOUBLE:
116  if (i_normtype == L2NORM) {
117  globalnormA = arma::sum(arma::square(m_Arows));
118  } else if (i_normtype == MAXNORM) {
119  globalnormA = arma::max(arma::square(m_Arows));
120  }
121  break;
122  case TWOD:
123  if (i_normtype == L2NORM) {
124  normc = arma::sum(arma::square(m_A));
125  MPI_Allreduce(normc.memptr(), globalnormA.memptr(), m_A.n_cols,
126  MPI_DOUBLE, MPI_SUM, this->m_mpicomm.commSubs()[1]);
127  } else if (i_normtype == MAXNORM) {
128  normc = arma::max(m_A);
129  MPI_Allreduce(normc.memptr(), globalnormA.memptr(), m_A.n_cols,
130  MPI_DOUBLE, MPI_SUM, this->m_mpicomm.commSubs()[1]);
131  }
132  break;
133  default:
134  INFO << "cannot normalize" << std::endl;
135  }
136  normmat = arma::repmat(globalnormA, m_A.n_rows, 1);
137  m_A /= normmat;
138  }

141  /*
142  * Uses the pattern from the input matrix X but
143  * the value is computed as low rank.
144  */
145  void randomLowRank(const UWORD m, const UWORD n, const UWORD k, MATTYPE* X) {
146  uint start_row = 0, end_row = 0, start_col = 0, end_col = 0;
147  switch (m_distio) {
148  case ONED_ROW:
149  start_row = MPI_RANK * (*X).n_rows;
150  end_row = ((MPI_RANK + 1) * (*X).n_rows) - 1;
151  start_col = 0;
152  end_col = (*X).n_cols - 1;
153  break;
154  case ONED_COL:
155  start_row = 0;
156  end_row = (*X).n_rows - 1;
157  start_col = MPI_RANK * (*X).n_cols;
158  end_col = ((MPI_RANK + 1) * (*X).n_cols) - 1;
159  break;
160  // in the case of ONED_DOUBLE we are stitching
161  // m/p x n/p matrices and in TWOD we are
162  // stitching m/pr * n/pc matrices.
163  case ONED_DOUBLE:
164  if ((*X).n_cols == n) { // m_Arows
165  start_row = MPI_RANK * (*X).n_rows;
166  end_row = ((MPI_RANK + 1) * (*X).n_rows) - 1;
167  start_col = 0;
168  end_col = n - 1;
169  }
170  if ((*X).n_rows == m) { // m_Acols
171  start_row = 0;
172  end_row = m - 1;
173  start_col = MPI_RANK * (*X).n_cols;
174  end_col = ((MPI_RANK + 1) * (*X).n_cols) - 1;
175  }
176  break;
177  case TWOD:
178  start_row = MPI_ROW_RANK * (*X).n_rows;
179  end_row = ((MPI_ROW_RANK + 1) * (*X).n_rows) - 1;
180  start_col = MPI_COL_RANK * (*X).n_cols;
181  end_col = ((MPI_COL_RANK + 1) * (*X).n_cols) - 1;
182  break;
183  }
184  // all machines will generate same Wrnd and Hrnd
185  // at all times.
186  arma::arma_rng::set_seed(kW_seed_idx);
187  MAT Wrnd(m, k);
188  Wrnd.randu();
189  MAT Hrnd(k, n);
190  Hrnd.randu();
191 #ifdef BUILD_SPARSE
192  SP_MAT::iterator start_it = (*X).begin();
193  SP_MAT::iterator end_it = (*X).end();
194  double tempVal = 0.0;
195  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
196  VEC Wrndi = vectorise(Wrnd.row(start_row + it.row()));
197  VEC Hrndj = Hrnd.col(start_col + it.col());
198  tempVal = dot(Wrndi, Hrndj);
199  (*it) = ceil(kalpha * tempVal + kbeta);
200  }
201 #else
202  MAT templr;
203  if ((*X).n_cols == n) { // ONED_ROW
204  MAT myWrnd = Wrnd.rows(start_row, end_row);
205  templr = myWrnd * Hrnd;
206  } else if ((*X).n_rows == m) { // ONED_COL
207  MAT myHcols = Hrnd.cols(start_col, end_col);
208  templr = Wrnd * myHcols;
209  } else if ((*X).n_rows == (m / MPI_SIZE) && (*X).n_cols == (n / MPI_SIZE) ||
210  ((*X).n_rows == (m / this->m_mpicomm.pr()) &&
211  (*X).n_cols == (n / this->m_mpicomm.pc()))) {
212  MAT myWrnd = Wrnd.rows(start_row, end_row);
213  MAT myHcols = Hrnd.cols(start_col, end_col);
214  templr = myWrnd * myHcols;
215  }
216  (*X) = ceil(kalpha * templr + kbeta);
217 #endif
218  }

220 #ifdef BUILD_SPARSE
221  void uniform_dist_matrix(MATTYPE& A) {
222  // make the matrix of ONED distribution
223  // sometimes in the pacoss it gives nearly equal
224  // distribution. we have to make sure everyone of
225  // same size;
226  int max_rows = 0, max_cols = 0;
227  int my_rows = A.n_rows;
228  int my_cols = A.n_cols;
229  bool last_exist = false;
230  double my_min_value = 0.0;
231  double overall_min;
232  UWORD my_correct_nnz = 0;
233  if (A.n_nonzero > 0) {
234  my_min_value = A.values[0];
235  }
236  MPI_Allreduce(&my_rows, &max_rows, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
237  MPI_Allreduce(&my_cols, &max_cols, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
238  // choose max rows and max cols nicely
239  max_rows -= (max_rows % m_mpicomm.pr());
240  max_cols -= (max_cols % m_mpicomm.pc());
241  // count the number of nnz's that satisfies this
242  // criteria
243  try {
244  if (A.n_nonzero > 0) {
245  SP_MAT::iterator start_it = A.begin();
246  SP_MAT::iterator end_it = A.end();
247  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
248  if (it.row() < max_rows && it.col() < max_cols) {
249  my_correct_nnz++;
250  if (*it != 0 && my_min_value < *it) {
251  my_min_value = *it;
252  }
253  }
254  if (it.row() == max_rows - 1 && it.col() == max_cols - 1) {
255  last_exist = true;
256  }
257  }
258  }
259  if (!last_exist) {
260  my_correct_nnz++;
261  }
262 
263  if (A.n_nonzero == 0) {
264  my_correct_nnz++;
265  }
266  MPI_Allreduce(&my_min_value, &overall_min, 1, MPI_INT, MPI_MIN,
267  MPI_COMM_WORLD);
268  arma::umat locs;
269  VEC vals;
270  DISTPRINTINFO("max_rows::" << max_rows << "::max_cols::" << max_cols
271  << "::my_rows::" << my_rows << "::my_cols::"
272  << my_cols << "::last_exist::" << last_exist
273  << "::my_nnz::" << A.n_nonzero
274  << "::my_correct_nnz::" << my_correct_nnz);
275  locs = arma::zeros<arma::umat>(2, my_correct_nnz);
276  vals = arma::zeros<VEC>(my_correct_nnz);
277  if (A.n_nonzero > 0) {
278  SP_MAT::iterator start_it = A.begin();
279  SP_MAT::iterator end_it = A.end();
280  double idx = 0;
281  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
282  if (it.row() < max_rows && it.col() < max_cols) {
283  locs(0, idx) = it.row();
284  locs(1, idx) = it.col();
285  vals(idx++) = *it;
286  }
287  }
288  } else {
289  locs(0, 0) = 0;
290  locs(1, 0) = 0;
291  vals(0) = overall_min;
292  }
293  if (A.n_nonzero > 0 && !last_exist) {
294  locs(0, my_correct_nnz - 1) = max_rows - 1;
295  locs(1, my_correct_nnz - 1) = max_cols - 1;
296  vals(my_correct_nnz - 1) = overall_min;
297  }
298  SP_MAT A_new(locs, vals);
299  A.clear();
300  A = A_new;
301  } catch (const std::exception& e) {
302  DISTPRINTINFO(e.what()
303  << "max_rows::" << max_rows << "::max_cols::" << max_cols
304  << "::my_rows::" << my_rows << "::my_cols::" << my_cols
305  << "::last_exist::" << last_exist << "::my_nnz::"
306  << A.n_nonzero << "::my_correct_nnz::" << my_correct_nnz);
307  }
308  }
309 #endif

314  /*
315  * We need m,n,pr,pc only for rand matrices. If otherwise we are
316  * expecting the file will hold all the details.
317  * If we are loading by file name we dont need distio flag.
318  *
319  */
320  void readInput(const std::string file_name, UWORD m = 0, UWORD n = 0,
321  UWORD k = 0, double sparsity = 0, UWORD pr = 0, UWORD pc = 0,
322  normtype i_normalization = NONE) {
323  // INFO << "readInput::" << file_name << "::" << distio << "::"
324  // << m << "::" << n << "::" << pr << "::" << pc
325  // << "::" << this->MPI_RANK << "::" << this->m_mpicomm.size() <<
326  // std::endl;
327  std::string rand_prefix("rand_");
328  if (!file_name.compare(0, rand_prefix.size(), rand_prefix)) {
329  std::string type = file_name.substr(rand_prefix.size());
330  assert(type == "normal" || type == "lowrank" || type == "uniform");
331  switch (m_distio) {
332  case ONED_ROW:
333  m_Arows.zeros(m / MPI_SIZE, n);
334  randMatrix(type, MPI_RANK + kPrimeOffset, sparsity, &m_Arows);
335  if (type == "lowrank") {
336  randomLowRank(m, n, k, &m_Arows);
337  }
338  break;
339  case ONED_COL:
340  m_Acols.zeros(m, n / MPI_SIZE);
341  randMatrix(type, MPI_RANK + kPrimeOffset, sparsity, &m_Acols);
342  if (type == "lowrank") {
343  randomLowRank(m, n, k, &m_Acols);
344  }
345  break;
346  case ONED_DOUBLE: {
347  int p = MPI_SIZE;
348  m_Arows.set_size(m / p, n);
349  m_Acols.set_size(m, n / p);
350  randMatrix(type, MPI_RANK + kPrimeOffset, sparsity, &m_Arows);
351  if (type == "lowrank") {
352  randomLowRank(m, n, k, &m_Arows);
353  }
354  randMatrix(type, MPI_RANK + kPrimeOffset, sparsity, &m_Acols);
355  if (type == "lowrank") {
356  randomLowRank(m, n, k, &m_Acols);
357  }
358  break;
359  }
360  case TWOD:
361  m_A.zeros(m / pr, n / pc);
362  randMatrix(type, MPI_RANK + kPrimeOffset, sparsity, &m_A);
363  if (type == "lowrank") {
364  randomLowRank(m, n, k, &m_A);
365  }
366  break;
367  }
368  } else {
369  std::stringstream sr, sc;
370  if (m_distio == ONED_ROW || m_distio == ONED_DOUBLE) {
371  sr << file_name << "rows_" << MPI_SIZE << "_" << MPI_RANK;
372 #ifdef BUILD_SPARSE
373  m_Arows.load(sr.str(), arma::coord_ascii);
374  uniform_dist_matrix(m_Arows);
375 #else
376  m_Arows.load(sr.str());
377 #endif
378  }
379  if (m_distio == ONED_COL || m_distio == ONED_DOUBLE) {
380  sc << file_name << "cols_" << MPI_SIZE << "_" << MPI_RANK;
381 #ifdef BUILD_SPARSE
382  m_Acols.load(sc.str(), arma::coord_ascii);
383  uniform_dist_matrix(m_Acols);
384 #else
385  m_Acols.load(sc.str());
386 #endif
387  m_Acols = m_Acols.t();
388  }
389  if (m_distio == TWOD) {
390  // sr << file_name << "_" << MPI_SIZE << "_" << MPI_RANK;
391  sr << file_name << MPI_RANK;
392 #ifdef BUILD_SPARSE
393  MAT temp_ijv;
394  temp_ijv.load(sr.str(), arma::raw_ascii);
395  if (temp_ijv.n_rows > 0 && temp_ijv.n_cols > 0) {
396  MAT vals(2, temp_ijv.n_rows);
397  MAT idxs_only = temp_ijv.cols(0, 1);
398  arma::umat idxs = arma::conv_to<arma::umat>::from(idxs_only);
399  arma::umat idxst = idxs.t();
400  vals = temp_ijv.col(2);
401  SP_MAT temp_spmat(idxst, vals);
402  m_A = temp_spmat;
403  } else {
404  arma::umat idxs = arma::zeros<arma::umat>(2, 1);
405  VEC vals = arma::zeros<VEC>(1);
406  SP_MAT temp_spmat(idxs, vals);
407  m_A = temp_spmat;
408  }
409  // m_A.load(sr.str(), arma::coord_ascii);
410  uniform_dist_matrix(m_A);
411 #else
412  m_A.load(sr.str());
413 #endif
414  }
415  }
416 #ifndef BUILD_SPARSE
417  if (i_normalization != NONE) {
418  normalize(i_normalization);
419  }
420 #endif
