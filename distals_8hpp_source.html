9 template <class INPUTMATTYPE>
10 class DistALS : public DistAUNMF<INPUTMATTYPE> {
11  protected:
12  // update W given HtH and AHt
13  void updateW() {
14  // AHtij is of size k*(globalm/p).
15  // this->W is of size (globalm/p)xk
16  // this->HtH is of size kxk
17  // w_ij = w_ij .* (AH)_ij/(WHtH)_ij
18  // Here ij is the element of W matrix.
19  this->Wt = arma::solve(arma::trimatl(this->HtH), this->AHtij);
20  this->W = this->Wt.t();
21  DISTPRINTINFO("ALS::updateW::HtH::" << PRINTMATINFO(this->HtH)
22  << "::AHtij::"
23  << PRINTMATINFO(this->AHtij)
24  << "::W::" << PRINTMATINFO(this->W));
25  DISTPRINTINFO("ALS::updateW::HtH::" << norm(this->HtH, "fro") << "::AHtij::"
26  << norm(this->AHtij, "fro")
27  << "::W::" << norm(this->W, "fro"));
28  }
29 
30  void updateH() {
31  // WtAij is of size k*(globaln/p)
32  // this->H is of size (globaln/p)xk
33  // this->WtW is of size kxk
34  // h_ij = h_ij .* WtAij.t()/(HWtW)_ij
35  // Here ij is the element of H matrix.
36  this->Ht = arma::solve(arma::trimatl(this->WtW), this->WtAij);
37  this->H = this->Ht.t();
38  DISTPRINTINFO("ALS::updateH::WtW::" << PRINTMATINFO(this->WtW)
39  << "::WtAij::"
40  << PRINTMATINFO(this->WtAij)
41  << "::H::" << PRINTMATINFO(this->H));
42  DISTPRINTINFO("ALS::updateH::WtW::" << norm(this->WtW, "fro") << "::WtAij::"
43  << norm(this->WtAij, "fro")
44  << "::H::" << norm(this->H, "fro"));
45  }
46 
47  public:
48  DistALS(const INPUTMATTYPE& input, const MAT& leftlowrankfactor,
49  const MAT& rightlowrankfactor, const MPICommunicator& communicator,
50  const int numkblks)
51  : DistAUNMF<INPUTMATTYPE>(input, leftlowrankfactor, rightlowrankfactor,
52  communicator, numkblks) {

54  }
