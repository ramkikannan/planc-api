17  void updateW() {
18  // AHtij is of size k*(globalm/p).
19  // this->W is of size (globalm/p)xk
20  // this->HtH is of size kxk
21  // Eq 14(a) page 7 of JGO paper
22  // W(:,i)=[W(:,i) + (AH(:,i)-W*HtH(:,i))/HtH(i,i)]_+
23  // column normalize W_i
24  // Here ij is the element of W matrix.
25  for (int i = 0; i < this->k; i++) {
26  // W(:,i) = max(W(:,i) * HHt_reg(i,i) + AHt(:,i) - W *
27  // HHt_reg(:,i),epsilon);
28  VEC updWi = this->W.col(i) * this->HtH(i, i) +
29  ((this->AHtij.row(i)).t() - this->W * this->HtH.col(i));
30 #ifdef MPI_VERBOSE
31  DISTPRINTINFO("b4 fixNumericalError::" << endl << updWi);
32 #endif // ifdef MPI_VERBOSE
33  fixNumericalError<VEC>(&updWi);
34 #ifdef MPI_VERBOSE
35  DISTPRINTINFO("after fixNumericalError::" << endl << updWi);
36 #endif // ifdef MPI_VERBOSE
37 
38  // W(:,i) = W(:,i)/norm(W(:,i));
39  double normWi = arma::norm(updWi, 2);
40  normWi *= normWi;
41  double globalnormWi;
42  mpitic();
43  MPI_Allreduce(&normWi, &globalnormWi, 1, MPI_DOUBLE, MPI_SUM,
44  MPI_COMM_WORLD);
45  double temp = mpitoc();
46  this->time_stats.communication_duration(temp);
47  this->time_stats.allreduce_duration(temp);
48 
49  if (globalnormWi > 0) {
50  this->W.col(i) = updWi / sqrt(globalnormWi);
51  this->H.col(i) = this->H.col(i) * sqrt(globalnormWi);
52  }
53  }
54  this->Wt = this->W.t();
55  }
56 
57  void updateH() {
58  // WtAij is of size k*(globaln/p)
59  // this->H is of size (globaln/p)xk
60  // this->WtW is of size kxk
61  // Eq 14(b) page 7 of JGO paper
62  // H(:,i) = H(:,i) + WtA(:,i) - H*WtW(:,i)
63  // Here ij is the element of H matrix.
64  for (int i = 0; i < this->k; i++) {
65  // H(i,:) = max(H(i,:) + WtA(i,:) - WtW_reg(i,:) * H,epsilon);
66  VEC updHi = this->H.col(i) +
67  ((this->WtAij.row(i)).t() - this->H * this->WtW.col(i));
68 #ifdef MPI_VERBOSE
69  DISTPRINTINFO("b4 fixNumericalError::" << endl << updHi);
70 #endif // ifdef MPI_VERBOSE
71  fixNumericalError<VEC>(&updHi);
72 #ifdef MPI_VERBOSE
73  DISTPRINTINFO("after fixNumericalError::" << endl << updHi);
74 #endif // ifdef MPI_VERBOSE
75  double normHi = arma::norm(updHi, 2);
76  normHi *= normHi;
77  double globalnormHi;
78  mpitic();
79  MPI_Allreduce(&normHi, &globalnormHi, 1, MPI_DOUBLE, MPI_SUM,
80  MPI_COMM_WORLD);
81  double temp = mpitoc();
82  this->time_stats.communication_duration(temp);
83  this->time_stats.allreduce_duration(temp);
84 
85  if (globalnormHi > 0) {
86  this->H.col(i) = updHi;
87  }
88  }
89  this->Ht = this->H.t();
90  }
