42  inline double get_alpha(double alpha, double q) {
43  /* Solves the quadratic equation
44  x^2 + (\alpha^2 -q)x - \alpha^2 = 0
45  */
46  double a, b, c, D, x;
47  a = 1.0;
48  b = alpha * alpha - q;
49  c = -alpha * alpha;
50  D = b * b - 4 * a * c;
51  x = (-b + sqrt(D)) / 2;
52  return x;
53  }

78  void accelerate() {
79  int iter = this->current_it();
80  if (iter > acc_start) {
81  int num_modes = m_prox_t.modes();
82  double cur_err = this->current_error();
83 
84  double acc_step = std::pow(iter + 1, (1.0 / acc_exp));
85  // Adjust all the factors
86  // Reusing gradient/acceleration term to save memory
87  m_grad_t.zeros();
88  int lowrank = m_prox_t.rank();
89  MAT scalecur = arma::eye<MAT>(lowrank, lowrank);
90  MAT scaleprev = arma::eye<MAT>(lowrank, lowrank);
91  for (int mode = 0; mode < num_modes; mode++) {
92  if (mode == num_modes - 1) {
93  scalecur = arma::diagmat(this->m_local_ncp_factors.lambda());
94  scaleprev = arma::diagmat(m_prev_t.lambda());
95  }
96  // Step Matrix
97  MAT acc_mat = m_grad_t.factor(mode);
98  acc_mat = (scalecur * this->m_local_ncp_factors_t.factor(mode)) -
99  (scaleprev * m_prev_t.factor(mode));
100  acc_mat *= acc_step;
101  acc_mat += (scalecur * this->m_local_ncp_factors_t.factor(mode));
102 
103  if (m_nls_sizes[mode] == 0) {
104  acc_mat.zeros();
105  }
106  m_acc_t.set(mode, acc_mat);
107  m_acc_t.distributed_normalize_rows(mode);
108  }
109  // Compute Error
110  // Always call with mode 0 to reuse MTTKRP if accepted
111  double acc_err = this->computeError(m_acc_t, 0);
112 
113  // Acceleration Accepted
114  if (acc_err < cur_err) {
115  // Set proximal term to current term
116  for (int mode = 0; mode < num_modes; mode++) {
117  m_prox_t.set(mode, m_acc_t.factor(mode));
118  m_prox_t.distributed_normalize_rows(mode);
119  }
120  PRINTROOT("Acceleration Successful::relative_error::" << acc_err);
121  } else {
122  // Acceleration Failed reset to prev iterate
123  this->reset(m_prox_t, true);
124  acc_fails++;
125  if (acc_fails > fail_limit) {
126  acc_fails = 0;
127  acc_exp++;
128  }
129  PRINTROOT("Acceleration Failed::relative_error::" << acc_err);
130  }
131  }
132  }

134  MAT update(const int mode) {
135  double L, mu, lambda, q, alpha, alpha_prev, beta;
136  int outer_iter = this->current_it();
137  int iter = 0;
138  if (mode == 0) {
139  m_prev_t.set_lambda(m_prox_t.lambda());
140  }
141  m_prev_t.set(mode, m_prox_t.factor(mode));
142  MAT Ht(m_prox_t.factor(mode));
143  MAT Htprev = Ht;
144  m_acc_t.set(mode, Ht);
145  modified_gram = this->global_gram;
146 
147  VEC eigval = arma::eig_sym(modified_gram);
148  L = eigval.max();
149  mu = eigval.min();
150  lambda = get_lambda(L, mu);
151  modified_gram.diag() += lambda;
152 
153  q = (mu + lambda) / (L + lambda);
154 
155  MAT modified_local_mttkrp_t =
156  (-1 * lambda) * m_acc_t.factor(mode) - this->ncp_local_mttkrp_t[mode];
157 
158  alpha = 1;
159  alpha_prev = 1;
160  beta = 1;
161 
162  while (true) {
163  m_grad_t.set(mode, modified_local_mttkrp_t +
164  (modified_gram * m_acc_t.factor(mode)));
165  if (stop_iter(mode)) break;
166 
167  iter++;
168  Htprev = Ht;
169  Ht = m_acc_t.factor(mode) - ((1 / (L + lambda)) * m_grad_t.factor(mode));
170  fixNumericalError<MAT>(&Ht, EPSILON_1EMINUS16);
171  Ht.for_each([](MAT::elem_type &val) { val = val > 0.0 ? val : 0.0; });
172  alpha_prev = alpha;
173  alpha = get_alpha(alpha_prev, q);
174  beta =
175  (alpha_prev * (1 - alpha_prev)) / (alpha + alpha_prev * alpha_prev);
176  m_acc_t.set(mode, Ht + beta * (Ht - Htprev));
177  }
178  if (m_nls_sizes[mode] == 0) {
179  Ht.zeros();
180  m_grad_t.set(mode, Ht);
181  }
182  PRINTROOT("Nesterov Update::mode::" << mode
183  << "::outer_iter::" << outer_iter
184  << "::NLS inner_iter::" << iter);
185  m_prox_t.set(mode, Ht);
186  m_prox_t.distributed_normalize_rows(mode);
187
