13 class NTFMPICommunicator {
14  private:
15  int m_global_rank;
16  int m_num_procs;
17  UVEC m_proc_grids;
18  MPI_Comm m_cart_comm;
19  // for mode communicators (*,...,p_n,...,*)
20  MPI_Comm *m_fiber_comm;
21  // all communicators other than the given mode.
22  // (p1,p2,...,p_n-1,*,p_n,...,p_M)
23  MPI_Comm *m_slice_comm;
24  UVEC m_fiber_ranks;
25  UVEC m_slice_ranks;
26  UVEC m_slice_sizes;
27  std::vector<int> m_coords;
28 
29  public:
30  void printConfig() {
31  if (rank() == 0) {
32  INFO << "successfully setup MPI communicators" << std::endl;
33  INFO << "size=" << size() << std::endl;
34  INFO << "processor grid size::" << m_proc_grids;
35  int slice_size;
36  MPI_Comm current_slice_comm;
37  for (int i = 0; i < MPI_CART_DIMS; i++) {
38  current_slice_comm = this->m_slice_comm[i];
39  MPI_Comm_size(current_slice_comm, &slice_size);
40  INFO << "Numprocs in slice " << i << "::" << slice_size << std::endl;
41  }
42  int fiber_size;
43  MPI_Comm current_fiber_comm;
44  for (int i = 0; i < MPI_CART_DIMS; i++) {
45  current_fiber_comm = this->m_fiber_comm[i];
46  MPI_Comm_size(current_fiber_comm, &fiber_size);
47  INFO << "Numprocs in fiber " << i << "::" << fiber_size << std::endl;
48  }
49  }
50  UVEC cooprint(MPI_CART_DIMS);
51  for (int ii = 0; ii < MPI_CART_DIMS; ii++) cooprint[ii] = m_coords[ii];
52 
53  MPI_Barrier(MPI_COMM_WORLD);
54  for (int i = 0; i < size(); i++) {
55  if (i == rank()) {
56  INFO << "slice ranks of rank::" << m_global_rank
57  << "::" << m_slice_ranks << std::endl;
58  INFO << "fiber ranks of rank::" << m_global_rank
59  << "::" << m_fiber_ranks << std::endl;
60  INFO << "coordinates::" << m_global_rank << "::" << cooprint
61  << std::endl;
62  }
63  MPI_Barrier(MPI_COMM_WORLD);
64  }
65  }

67  NTFMPICommunicator(int argc, char *argv[], const UVEC &i_dims)
68  : m_proc_grids(i_dims) {
69  // Get the number of MPI processes
70  MPI_Init(&argc, &argv);
71  MPI_Comm_rank(MPI_COMM_WORLD, &m_global_rank);
72  MPI_Comm_size(MPI_COMM_WORLD, &m_num_procs);
73  int grid_count = m_proc_grids[0];
74  if (m_num_procs != arma::prod(m_proc_grids)) {
75  ERR << "number of mpi process and process grid doesn't match";
76  MPI_Barrier(MPI_COMM_WORLD);
77  MPI_Abort(MPI_COMM_WORLD, 1);
78  exit(-1);
79  }
80  // Create a virtual topology MPI communicator
81  std::vector<int> periods(MPI_CART_DIMS);
82  std::vector<int> m_proc_grids_vec =
83  arma::conv_to<std::vector<int>>::from(m_proc_grids);
84  for (int i = 0; i < MPI_CART_DIMS; i++) periods[i] = 1;
85  int reorder = 0;
86  MPI_Cart_create(MPI_COMM_WORLD, MPI_CART_DIMS, &m_proc_grids_vec[0],
87  &periods[0], reorder, &m_cart_comm);
88 
89  // Allocate memory for subcommunicators
90  m_fiber_comm = new MPI_Comm[MPI_CART_DIMS];
91  m_slice_comm = new MPI_Comm[MPI_CART_DIMS];
92 
93  // Get the subcommunicators
94  std::vector<int> remainDims(MPI_CART_DIMS);
95  for (int i = 0; i < remainDims.size(); i++) remainDims[i] = 1;
96  // initialize the fiber ranks
97  m_slice_ranks = arma::zeros<UVEC>(MPI_CART_DIMS);
98  int current_slice_rank;
99  for (int i = 0; i < MPI_CART_DIMS; i++) {
100  remainDims[i] = 0;
101  MPI_Cart_sub(m_cart_comm, &remainDims[0], &(m_slice_comm[i]));
102  remainDims[i] = 1;
103  MPI_Comm_rank(m_slice_comm[i], &current_slice_rank);
104  m_slice_ranks[i] = current_slice_rank;
105  }
106  for (int i = 0; i < remainDims.size(); i++) remainDims[i] = 0;
107  m_fiber_ranks = arma::zeros<UVEC>(MPI_CART_DIMS);
108  int current_fiber_rank;
109  for (int i = 0; i < MPI_CART_DIMS; i++) {
110  remainDims[i] = 1;
111  MPI_Cart_sub(m_cart_comm, &remainDims[0], &(m_fiber_comm[i]));
112  remainDims[i] = 0;
113  MPI_Comm_rank(m_fiber_comm[i], &current_fiber_rank);
114  m_fiber_ranks[i] = current_fiber_rank;
115  }
116  // Get the coordinates
117  m_coords.resize(MPI_CART_DIMS, 0);
118  MPI_Cart_coords(m_cart_comm, m_global_rank, MPI_CART_DIMS, &m_coords[0]);
119  // Get the slice size
120  m_slice_sizes = arma::zeros<UVEC>(MPI_CART_DIMS);
121  for (int i = 0; i < MPI_CART_DIMS; i++) {
122  m_slice_sizes[i] = m_num_procs / m_proc_grids[i];
123  }
124  }
125 
126  ~NTFMPICommunicator() {
127  MPI_Barrier(MPI_COMM_WORLD);
128  int finalized;
129  MPI_Finalized(&finalized);
130 
131  if (!finalized) {
132  for (int i = 0; i < MPI_CART_DIMS; i++) {
133  MPI_Comm_free(&m_fiber_comm[i]);
134  MPI_Comm_free(&m_slice_comm[i]);
135  }
136  }
137  delete[] m_fiber_comm;
138  delete[] m_slice_comm;
139  }

159  return n_procs;
160  }
161  int fiber_rank(int i) const { return m_fiber_ranks[i]; }
162  int slice_rank(int i) const { return m_slice_ranks[i]; }
163  UVEC proc_grids() const { return this->m_proc_grids; }
164  int rank() const { return m_global_rank; }
165  int num_slices(int mode) const { return m_proc_grids[mode]; }
166  int slice_num(int mode) const { return m_coords[mode]; }
167  int slice_size(int mode) const { return m_slice_sizes[mode]; }
168 
169  /*
170  * Return true only for those processors whose
171  * coordinates are non-zero for the mode and zero
172  * non-modes
173  */
174  bool isparticipating(int mode) {
175  bool rc = true;
176  size_t num_modes = this->m_proc_grids.n_rows;
177  for (int i = 0; i < num_modes; i++) {
178  if (i != mode && this->m_coords[i] != 0) {
179  rc = false;
180  break;
181  }
182  }
183  return rc;
184  }
