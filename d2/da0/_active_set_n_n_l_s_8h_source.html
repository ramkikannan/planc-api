42 /* Algorithm NNLS: NONNEGATIVE LEAST SQUARES */
43 
44 /* The original version of this code was developed by */
45 /* Charles L. Lawson and Richard J. Hanson at Jet Propulsion Laboratory */
46 /* 1973 JUN 15, and published in the book */
47 /* "SOLVING LEAST SQUARES PROBLEMS", Prentice-HalL, 1974. */
48 /* Revised FEB 1995 to accompany reprinting of the book by SIAM. */
49 
50 /* GIVEN AN M BY N MATRIX, A, AND AN M-VECTOR, B, COMPUTE AN */
51 /* N-VECTOR, X, THAT SOLVES THE LEAST SQUARES PROBLEM */
52 
53 /* A * X = B SUBJECT TO X .GE. 0 */
54 /* ------------------------------------------------------------------ */
55 /* Subroutine Arguments */
56 
57 /* A(),MDA,M,N MDA IS THE FIRST DIMENSIONING PARAMETER FOR THE */
58 /* ARRAY, A(). ON ENTRY A() CONTAINS THE M BY N */
59 /* MATRIX, A. ON EXIT A() CONTAINS */
60 /* THE PRODUCT MATRIX, Q*A , WHERE Q IS AN */
61 /* M BY M ORTHOGONAL MATRIX GENERATED IMPLICITLY BY */
62 /* THIS SUBROUTINE. */
63 /* B() ON ENTRY B() CONTAINS THE M-VECTOR, B. ON EXIT B() CON- */
64 /* TAINS Q*B. */
65 /* X() ON ENTRY X() NEED NOT BE INITIALIZED. ON EXIT X() WILL */
66 /* CONTAIN THE SOLUTION VECTOR. */
67 /* RNORM ON EXIT RNORM CONTAINS THE EUCLIDEAN NORM OF THE */
68 /* RESIDUAL VECTOR. */
69 /* W() AN N-ARRAY OF WORKING SPACE. ON EXIT W() WILL CONTAIN */
70 /* THE DUAL SOLUTION VECTOR. W WILL SATISFY W(I) = 0. */
71 /* FOR ALL I IN SET P AND W(I) .LE. 0. FOR ALL I IN SET Z */
72 /* ZZ() AN M-ARRAY OF WORKING SPACE. */
73 /* INDEX() AN INT WORKING ARRAY OF LENGTH AT LEAST N. */
74 /* ON EXIT THE CONTENTS OF THIS ARRAY DEFINE THE SETS */
75 /* P AND Z AS FOLLOWS.. */
76 
77 /* INDEX(1) THRU INDEX(NSETP) = SET P. */
78 /* INDEX(IZ1) THRU INDEX(IZ2) = SET Z. */
79 /* IZ1 = NSETP + 1 = NPP1 */
80 /* IZ2 = N */
81 /* MODE THIS IS A SUCCESS-FAILURE FLAG WITH THE FOLLOWING */
82 /* MEANINGS. */
83 /* 1 THE SOLUTION HAS BEEN COMPUTED SUCCESSFULLY. */
84 /* 2 THE DIMENSIONS OF THE PROBLEM ARE BAD. */
85 /* EITHER M .LE. 0 OR N .LE. 0. */
86 /* 3 ITERATION COUNT EXCEEDED. MORE THAN 3*N ITERATIONS. */

97 
98 /* CONSTRUCTION AND/OR APPLICATION OF A SINGLE */
99 /* HOUSEHOLDER TRANSFORMATION.. Q = I + U*(U**T)/B */
100 
101 /* The original version of this code was developed by */
102 /* Charles L. Lawson and Richard J. Hanson at Jet Propulsion Laboratory */
103 /* 1973 JUN 12, and published in the book */
104 /* "SOLVING LEAST SQUARES PROBLEMS", Prentice-HalL, 1974. */
105 /* Revised FEB 1995 to accompany reprinting of the book by SIAM. */
106 /* ------------------------------------------------------------------ */
107 /* Subroutine Arguments */
108 
109 /* MODE = 1 OR 2 Selects Algorithm H1 to construct and apply a */
110 /* Householder transformation, or Algorithm H2 to apply a */
111 /* previously constructed transformation. */
112 /* LPIVOT IS THE INDEX OF THE PIVOT ELEMENT. */
113 /* L1,M IF L1 .LE. M THE TRANSFORMATION WILL BE CONSTRUCTED TO */
114 /* ZERO ELEMENTS INDEXED FROM L1 THROUGH M. IF L1 GT. M */
115 /* THE SUBROUTINE DOES AN IDENTITY TRANSFORMATION. */
116 /* U(),IUE,UP On entry with MODE = 1, U() contains the pivot */
117 /* vector. IUE is the storage increment between elements. */
118 /* On exit when MODE = 1, U() and UP contain quantities */
119 /* defining the vector U of the Householder transformation. */
120 /* on entry with MODE = 2, U() and UP should contain */
121 /* quantities previously computed with MODE = 1. These will */
122 /* not be modified during the entry with MODE = 2. */
123 /* C() ON ENTRY with MODE = 1 or 2, C() CONTAINS A MATRIX WHICH */
124 /* WILL BE REGARDED AS A SET OF VECTORS TO WHICH THE */
125 /* HOUSEHOLDER TRANSFORMATION IS TO BE APPLIED. */
126 /* ON EXIT C() CONTAINS THE SET OF TRANSFORMED VECTORS. */
127 /* ICE STORAGE INCREMENT BETWEEN ELEMENTS OF VECTORS IN C(). */
128 /* ICV STORAGE INCREMENT BETWEEN VECTORS IN C(). */
129 /* NCV NUMBER OF VECTORS IN C() TO BE TRANSFORMED. IF NCV .LE. 0 */
130 /* NO OPERATIONS WILL BE DONE ON C(). */

161  ActiveSetNNLS(int rows, int maxCols) :

164  _workA = new T[_rows * _maxCols];

172  ~ActiveSetNNLS() {

174  delete[] _workW;
175  delete[] _workZ;
176  delete[] _workIndex;

179  bool solve(T* A, int numCols, T* b, T* x, T& rNorm) {
180  int workMode;
181 
182  // make copy of A, since nnls() clobbers it
183  memcpy(_workA, A, numCols * _rows * sizeof(T));
184 
185  // do it
186  nnls(_workA, _rows, _rows, numCols, b, x, &rNorm,
187  _workW, _workZ, _workIndex, &workMode, _maxIter);
188 
189  return (workMode == 1);
190  }
191 
192  int& maxIter() { return _maxIter;}
