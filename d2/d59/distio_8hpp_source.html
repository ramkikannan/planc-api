145  void randomLowRank(const UWORD m, const UWORD n, const UWORD k, MATTYPE* X) {
146  uint start_row = 0, end_row = 0, start_col = 0, end_col = 0;
147  switch (m_distio) {
148  case ONED_ROW:
149  start_row = MPI_RANK * (*X).n_rows;
150  end_row = ((MPI_RANK + 1) * (*X).n_rows) - 1;
151  start_col = 0;
152  end_col = (*X).n_cols - 1;
153  break;
154  case ONED_COL:
155  start_row = 0;
156  end_row = (*X).n_rows - 1;
157  start_col = MPI_RANK * (*X).n_cols;
158  end_col = ((MPI_RANK + 1) * (*X).n_cols) - 1;
159  break;
160  // in the case of ONED_DOUBLE we are stitching
161  // m/p x n/p matrices and in TWOD we are
162  // stitching m/pr * n/pc matrices.
163  case ONED_DOUBLE:
164  if ((*X).n_cols == n) { // m_Arows
165  start_row = MPI_RANK * (*X).n_rows;
166  end_row = ((MPI_RANK + 1) * (*X).n_rows) - 1;
167  start_col = 0;
168  end_col = n - 1;
169  }
170  if ((*X).n_rows == m) { // m_Acols
171  start_row = 0;
172  end_row = m - 1;
173  start_col = MPI_RANK * (*X).n_cols;
174  end_col = ((MPI_RANK + 1) * (*X).n_cols) - 1;
175  }
176  break;
177  case TWOD:
178  start_row = MPI_ROW_RANK * (*X).n_rows;
179  end_row = ((MPI_ROW_RANK + 1) * (*X).n_rows) - 1;
180  start_col = MPI_COL_RANK * (*X).n_cols;
181  end_col = ((MPI_COL_RANK + 1) * (*X).n_cols) - 1;
182  break;
183  }
184  // all machines will generate same Wrnd and Hrnd
185  // at all times.
186  arma::arma_rng::set_seed(kW_seed_idx);
187  MAT Wrnd(m, k);
188  Wrnd.randu();
189  MAT Hrnd(k, n);
190  Hrnd.randu();
191 #ifdef BUILD_SPARSE
192  SP_MAT::iterator start_it = (*X).begin();
193  SP_MAT::iterator end_it = (*X).end();
194  double tempVal = 0.0;
195  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
196  VEC Wrndi = vectorise(Wrnd.row(start_row + it.row()));
197  VEC Hrndj = Hrnd.col(start_col + it.col());
198  tempVal = dot(Wrndi, Hrndj);
199  (*it) = ceil(kalpha * tempVal + kbeta);
200  }
201 #else
202  MAT templr;
203  if ((*X).n_cols == n) { // ONED_ROW
204  MAT myWrnd = Wrnd.rows(start_row, end_row);
205  templr = myWrnd * Hrnd;
206  } else if ((*X).n_rows == m) { // ONED_COL
207  MAT myHcols = Hrnd.cols(start_col, end_col);
208  templr = Wrnd * myHcols;
209  } else if ((*X).n_rows == (m / MPI_SIZE) && (*X).n_cols == (n / MPI_SIZE) ||
210  ((*X).n_rows == (m / this->m_mpicomm.pr()) &&
211  (*X).n_cols == (n / this->m_mpicomm.pc()))) {
212  MAT myWrnd = Wrnd.rows(start_row, end_row);
213  MAT myHcols = Hrnd.cols(start_col, end_col);
214  templr = myWrnd * myHcols;
215  }
216  (*X) = ceil(kalpha * templr + kbeta);
217 #endif
218  }

221  void uniform_dist_matrix(MATTYPE& A) {
222  // make the matrix of ONED distribution
223  // sometimes in the pacoss it gives nearly equal
224  // distribution. we have to make sure everyone of
225  // same size;
226  int max_rows = 0, max_cols = 0;
227  int my_rows = A.n_rows;
228  int my_cols = A.n_cols;
229  bool last_exist = false;
230  double my_min_value = 0.0;
231  double overall_min;
232  UWORD my_correct_nnz = 0;
233  if (A.n_nonzero > 0) {
234  my_min_value = A.values[0];
235  }
236  MPI_Allreduce(&my_rows, &max_rows, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
237  MPI_Allreduce(&my_cols, &max_cols, 1, MPI_INT, MPI_MAX, MPI_COMM_WORLD);
238  // choose max rows and max cols nicely
239  max_rows -= (max_rows % m_mpicomm.pr());
240  max_cols -= (max_cols % m_mpicomm.pc());
241  // count the number of nnz's that satisfies this
242  // criteria
243  try {
244  if (A.n_nonzero > 0) {
245  SP_MAT::iterator start_it = A.begin();
246  SP_MAT::iterator end_it = A.end();
247  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
248  if (it.row() < max_rows && it.col() < max_cols) {
249  my_correct_nnz++;
250  if (*it != 0 && my_min_value < *it) {
251  my_min_value = *it;
252  }
253  }
254  if (it.row() == max_rows - 1 && it.col() == max_cols - 1) {
255  last_exist = true;
256  }
257  }
258  }
259  if (!last_exist) {
260  my_correct_nnz++;
261  }
262 
263  if (A.n_nonzero == 0) {
264  my_correct_nnz++;
265  }
266  MPI_Allreduce(&my_min_value, &overall_min, 1, MPI_INT, MPI_MIN,
267  MPI_COMM_WORLD);
268  arma::umat locs;
269  VEC vals;
270  DISTPRINTINFO("max_rows::" << max_rows << "::max_cols::" << max_cols
271  << "::my_rows::" << my_rows << "::my_cols::"
272  << my_cols << "::last_exist::" << last_exist
273  << "::my_nnz::" << A.n_nonzero
274  << "::my_correct_nnz::" << my_correct_nnz);
275  locs = arma::zeros<arma::umat>(2, my_correct_nnz);
276  vals = arma::zeros<VEC>(my_correct_nnz);
277  if (A.n_nonzero > 0) {
278  SP_MAT::iterator start_it = A.begin();
279  SP_MAT::iterator end_it = A.end();
280  double idx = 0;
281  for (SP_MAT::iterator it = start_it; it != end_it; ++it) {
282  if (it.row() < max_rows && it.col() < max_cols) {
283  locs(0, idx) = it.row();
284  locs(1, idx) = it.col();
285  vals(idx++) = *it;
286  }
287  }
288  } else {
289  locs(0, 0) = 0;
290  locs(1, 0) = 0;
291  vals(0) = overall_min;
292  }
293  if (A.n_nonzero > 0 && !last_exist) {
294  locs(0, my_correct_nnz - 1) = max_rows - 1;
295  locs(1, my_correct_nnz - 1) = max_cols - 1;
296  vals(my_correct_nnz - 1) = overall_min;
297  }
298  SP_MAT A_new(locs, vals);
299  A.clear();
300  A = A_new;
301  } catch (const std::exception& e) {
302  DISTPRINTINFO(e.what()
303  << "max_rows::" << max_rows << "::max_cols::" << max_cols
304  << "::my_rows::" << my_rows << "::my_cols::" << my_cols
305  << "::last_exist::" << last_exist << "::my_nnz::"
306  << A.n_nonzero << "::my_correct_nnz::" << my_correct_nnz);
307  }
308  }
