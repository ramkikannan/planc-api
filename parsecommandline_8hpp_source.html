52  void parseArrayofString(const char opt, const char *input) {
53  std::stringstream ss(input);
54  std::string s;
55  int i = 0;
56  if (this->m_num_modes == 0) {
57  while (getline(ss, s, ' ')) {
58  i++;
59  }
60  this->m_num_modes = i;
61  this->m_dimensions = arma::zeros<UVEC>(this->m_num_modes);
62  this->m_regularizers = arma::zeros<FVEC>(2 * this->m_num_modes);
63  this->m_proc_grids = arma::ones<UVEC>(this->m_num_modes);
64  }
65  i = 0;
66  ss.clear();
67  ss.str(input);
68  while (getline(ss, s, ' ')) {
69  switch (opt) {
70  case 'd':
71  this->m_dimensions(i++) = ::atoi(s.c_str());
72  break;
73  case 'r':
74  this->m_regularizers(i++) = ::atof(s.c_str());
75  break;
76  case 'p':
77  this->m_proc_grids(i++) = ::atoi(s.c_str());
78  break;
79  default:
80  INFO << "wrong option::" << opt << "::values::" << input << std::endl;
81  }
82  }
83  }

101  void parseplancopts() {
102  int opt, long_index;
103  while ((opt = getopt_long(this->m_argc, this->m_argv,
104  "a:d:e:i:k:o:p:r:s:t:", plancopts,
105  &long_index)) != -1) {
106  switch (opt) {
107  case 'a':
108  this->m_lucalgo = static_cast<algotype>(atoi(optarg));
109  break;
110  case 'e':
111  this->m_compute_error = atoi(optarg);
112  break;
113  case 'i': {
114  std::string temp = std::string(optarg);
115  this->m_Afile_name = temp;
116  break;
117  }
118  case 'k':
119  this->m_k = atoi(optarg);
120  break;
121  case 'o': {
122  std::string temp = std::string(optarg);
123  this->m_outputfile_name = temp;
124  break;
125  }
126  case 'd':
127  case 'r':
128  case 'p':
129  parseArrayofString(opt, optarg);
130  break;
131  case 's':
132  this->m_sparsity = atof(optarg);
133  break;
134  case 't':
135  this->m_num_it = atoi(optarg);
136  break;
137  case NUMKBLOCKS:
138  this->m_num_k_blocks = atoi(optarg);
139  break;
140  case NORMALIZATION: {
141  std::string temp = std::string(optarg);
142  if (temp.compare("l2") == 0) {
143  this->m_input_normalization = normtype::L2NORM;
144  } else if (temp.compare("max") == 0) {
145  this->m_input_normalization = normtype::MAXNORM;
146  }
147  break;
148  }
149  case DIMTREE:
150  this->m_dim_tree = atoi(optarg);
151  break;
152  default:
153  std::cout << "failed while processing argument:" << optarg
154  << std::endl;
155  print_usage();
156  exit(EXIT_FAILURE);
157  }
158  }
159  // properly initialize the values for nmf cases now.
160  if (this->m_num_modes == 2) {
161  this->m_globalm = this->m_dimensions(0);
162  this->m_globaln = this->m_dimensions(1);
163  this->m_regW(0) = this->m_regularizers(0);
164  this->m_regW(1) = this->m_regularizers(1);
165  this->m_regH(0) = this->m_regularizers(2);
166  this->m_regH(1) = this->m_regularizers(3);
167  this->m_pr = this->m_proc_grids(0);
168  this->m_pc = this->m_proc_grids(1);
169  }
170  }
