329  // Distribution normalization of factor matrices

331  void distributed_normalize() {
332  double local_colnorm;
333  double global_colnorm;
334  for (int i = 0; i < this->m_modes; i++) {
335  for (int j = 0; j < this->m_k; j++) {
336  local_colnorm = arma::norm(this->ncp_factors[i].col(j));
337  local_colnorm *= local_colnorm;
338  MPI_Allreduce(&local_colnorm, &global_colnorm, 1, MPI_DOUBLE, MPI_SUM,
339  MPI_COMM_WORLD);
340  global_colnorm = std::sqrt(global_colnorm);
341  if (global_colnorm > 0) this->ncp_factors[i].col(j) /= global_colnorm;
342  m_lambda(j) *= global_colnorm;
343  }
344  }
