13 /*
14  * There are totally prxpc process.
15  * Each process will hold the following
16  * An A of size (globalm/pr) x (globaln/pc)
17  * Here each process m=globalm/pr and n=globaln/pc
18  * H of size (globaln/p)xk
19  * W of size (globalm/p)xk
20  * A is mxn matrix
21  * H is nxk matrix
22  */

179  /*
180  * This is a matrix multiplication routine based on
181  * reduce_scatter.
182  * A is mxn in column major ordering
183  * W is mxk in column major ordering
184  * AtW is nxk in column major ordering
185  * There are totally p processes. Every process has
186  * A_i as m_i * n
187  * W_i as m_i * k
188  * AtW_i as n_i * k
189  * this->m_mpicomm.comm_subs()[0] is column communicator.
190  * this->m_mpicomm.comm_subs()[1] is row communicator.
191  */
192  void distWtA() {
193  for (int i = 0; i < num_k_blocks; i++) {
194  int start_row = i * perk;
195  int end_row = (i + 1) * perk - 1;
196  Wt_blk = Wt.rows(start_row, end_row);
197  distWtABlock();
198  WtAij.rows(start_row, end_row) = WtAij_blk;
199  }
200  }
201  void distWtABlock() {
202 #ifdef USE_PACOSS
203  // Perform expand communication using Pacoss.
204  memcpy(Wit.memptr(), Wt_blk.memptr(),
205  Wt_blk.n_rows * Wt_blk.n_cols * sizeof(Wt_blk[0]));
206  MPITIC;
207  this->m_rowcomm->expCommBegin(Wit.memptr(), this->perk);
208  this->m_rowcomm->expCommFinish(Wit.memptr(), this->perk);
209 #else
210  int sendcnt = (this->globalm() / MPI_SIZE) * this->perk;
211  int recvcnt = (this->globalm() / MPI_SIZE) * this->perk;
212  Wit.zeros();
213  MPITIC; // allgather WtA
214  MPI_Allgather(Wt_blk.memptr(), sendcnt, MPI_DOUBLE, Wit.memptr(), recvcnt,
215  MPI_DOUBLE, this->m_mpicomm.commSubs()[1]);
216 #endif
217  double temp = MPITOC; // allgather WtA
218  PRINTROOT("n::" << this->n << "::k::" << this->k << PRINTMATINFO(Wt)
219  << PRINTMATINFO(Wit));
220 #ifdef MPI_VERBOSE
221  DISTPRINTINFO(PRINTMAT(Wt_blk));
222  DISTPRINTINFO(PRINTMAT(Wit));
223 #endif
224  this->time_stats.communication_duration(temp);
225  this->time_stats.allgather_duration(temp);
226  MPITIC; // mm WtA
227  this->WitAij = this->Wit * this->A;
228  // #if defined(MKL_FOUND) && defined(BUILD_SPARSE)
229  // // void ARMAMKLSCSCMM(const SRC &mklMat, const DESTN &Bt, const char
230  // transa,
231  // // DESTN *Ct)
232  // ARMAMKLSCSCMM(this->A_ij_t, 'N', this->Wit, this->AijWit.memptr());
233  // #ifdef MPI_VERBOSE
234  // DISTPRINTINFO(PRINTMAT(this->AijWit));
235  // #endif
236  // this->WitAij = reshape(this->AijWit, this->k, this->n);
237  // #else
238  // this->WitAij = this->Wit * this->A;
239  // #endif
240  temp = MPITOC; // mm WtA
241 #ifdef MPI_VERBOSE
242  DISTPRINTINFO(PRINTMAT(this->WitAij));
243 #endif
244  PRINTROOT(PRINTMATINFO(this->A)
245  << PRINTMATINFO(this->Wit) << PRINTMATINFO(this->WitAij));
246  this->time_stats.compute_duration(temp);
247  this->time_stats.mm_duration(temp);
248  this->reportTime(temp, "WtA::");
249 #ifdef USE_PACOSS
250  // Perform fold communication using Pacoss.
251  MPITIC;
252  this->m_colcomm->foldCommBegin(WitAij.memptr(), this->perk);
253  this->m_colcomm->foldCommFinish(WitAij.memptr(), this->perk);
254  temp = MPITOC;
255  memcpy(WtAij_blk.memptr(), WitAij.memptr(),
256  WtAij_blk.n_rows * WtAij_blk.n_cols * sizeof(WtAij_blk[0]));
257 #else
258  WtAij_blk.zeros();
259  MPITIC; // reduce_scatter WtA
260  MPI_Reduce_scatter(this->WitAij.memptr(), this->WtAij_blk.memptr(),
261  &(this->recvWtAsize[0]), MPI_DOUBLE, MPI_SUM,
262  this->m_mpicomm.commSubs()[0]);
263  temp = MPITOC; // reduce_scatter WtA
264 #endif
265  this->time_stats.communication_duration(temp);
266  this->time_stats.reducescatter_duration(temp);
267  }
268  /*
269  * There are totally prxpc process.
270  * Each process will hold the following
271  * An A of size (m/pr) x (n/pc)
272  * H of size (n/p)xk
273  * find AHt kx(m/p) by reducing and scatter it using MPI_Reduce_scatter call.
274  * That is, p process will hold a kx(m/p) matrix.
275  * this->m_mpicomm.comm_subs()[0] is column communicator.
276  * this->m_mpicomm.comm_subs()[1] is row communicator.
277  * To preserve the memory for Hj, we collect only partial k
278  */
279  void distAH() {
280  for (int i = 0; i < num_k_blocks; i++) {
281  int start_row = i * perk;
282  int end_row = (i + 1) * perk - 1;
283  Ht_blk = Ht.rows(start_row, end_row);
284  distAHBlock();
285  AHtij.rows(start_row, end_row) = AHtij_blk;
286  }
287  }
288  void distAHBlock() {
289  /*
290  DISTPRINTINFO("distAH::" << "::Acolst::" \
291  Acolst.n_rows<<"x"<<Acolst.n_cols \
292  << "::norm::" << arma::norm(Acolst, "fro"));
293  DISTPRINTINFO("distAH::" << "::H::" \
294  << this->H.n_rows << "x" << this->H.n_cols);
295  */
296 #ifdef USE_PACOSS
297  // Perform expand communication using Pacoss.
298  memcpy(Hjt.memptr(), Ht_blk.memptr(),
299  Ht_blk.n_rows * Ht_blk.n_cols * sizeof(Ht_blk[0]));
300  MPITIC;
301  this->m_colcomm->expCommBegin(Hjt.memptr(), this->perk);
302  this->m_colcomm->expCommFinish(Hjt.memptr(), this->perk);
303 #else
304  int sendcnt = (this->globaln() / MPI_SIZE) * this->perk;
305  int recvcnt = (this->globaln() / MPI_SIZE) * this->perk;
306  Hjt.zeros();
307  MPITIC; // allgather AH
308  MPI_Allgather(this->Ht_blk.memptr(), sendcnt, MPI_DOUBLE,
309  this->Hjt.memptr(), recvcnt, MPI_DOUBLE,
310  this->m_mpicomm.commSubs()[0]);
311 #endif
312  PRINTROOT("n::" << this->n << "::k::" << this->k << PRINTMATINFO(Ht)
313  << PRINTMATINFO(Hjt));
314  double temp = MPITOC; // allgather AH
315 #ifdef MPI_VERBOSE
316  DISTPRINTINFO(PRINTMAT(Ht_blk));
317  DISTPRINTINFO(PRINTMAT(Hjt));
318  DISTPRINTINFO(PRINTMAT(this->A_ij_t));
319 #endif
320  this->time_stats.communication_duration(temp);
321  this->time_stats.allgather_duration(temp);
322  MPITIC; // mm AH
323  this->AijHjt = this->Hjt * this->A_ij_t;
324  // #if defined(MKL_FOUND) && defined(BUILD_SPARSE)
325  // // void ARMAMKLSCSCMM(const SRC &mklMat, const DESTN &Bt, const char
326  // transa,
327  // // DESTN *Ct)
328  // ARMAMKLSCSCMM(this->A, 'N', this->Hjt, this->AijHj.memptr());
329  // #ifdef MPI_VERBOSE
330  // DISTPRINTINFO(PRINTMAT(this->AijHj));
331  // #endif
332  // this->AijHjt = reshape(this->AijHj, this->k, this->m);
333  // #else
334  // this->AijHjt = this->Hjt * this->A_ij_t;
335  // #endif
336  // #ifdef MPI_VERBOSE
337  // DISTPRINTINFO(PRINTMAT(this->AijHjt));
338  // #endif
339  temp = MPITOC; // mm AH
340  PRINTROOT(PRINTMATINFO(this->A_ij_t)
341  << PRINTMATINFO(this->Hjt) << PRINTMATINFO(this->AijHjt));
342  this->time_stats.compute_duration(temp);
343  this->time_stats.mm_duration(temp);
344  this->reportTime(temp, "AH::");
345 #ifdef USE_PACOSS
346  // Perform fold communication using Pacoss.
347  MPITIC;
348  this->m_rowcomm->foldCommBegin(AijHjt.memptr(), this->perk);
349  this->m_rowcomm->foldCommFinish(AijHjt.memptr(), this->perk);
350  temp = MPITOC;
351  memcpy(AHtij_blk.memptr(), AijHjt.memptr(),
352  AHtij_blk.n_rows * AHtij_blk.n_cols * sizeof(AHtij_blk[0]));
353 #else
354  AHtij_blk.zeros();
355  MPITIC; // reduce_scatter AH
356  MPI_Reduce_scatter(this->AijHjt.memptr(), this->AHtij_blk.memptr(),
357  &(this->recvAHsize[0]), MPI_DOUBLE, MPI_SUM,
358  this->m_mpicomm.commSubs()[1]);
359  temp = MPITOC; // reduce_scatter AH
360 #endif
361  this->time_stats.communication_duration(temp);
362  this->time_stats.reducescatter_duration(temp);
363  }
364  /*
365  * There are p processes.
366  * Every process i has W in m_i * k
367  * At the end of this call, all process will have
368  * WtW of size k*k is symmetric. So not to worry
369  * about column/row major formats.
370  */
371  void distInnerProduct(const MAT &X, MAT *XtX) {
372  // each process computes its own kxk matrix
373  MPITIC; // gram
374  localWtW = X.t() * X;
375 #ifdef MPI_VERBOSE
376  DISTPRINTINFO("W::" << norm(X, "fro")
377  << "::localWtW::" << norm(this->localWtW, "fro"));
378 #endif
379  double temp = MPITOC; // gram
380  this->time_stats.compute_duration(temp);
381  this->time_stats.gram_duration(temp);
382  (*XtX).zeros();
383  if (X.n_rows == this->m) {
384  this->reportTime(temp, "Gram::W::");
385  } else {
386  this->reportTime(temp, "Gram::H::");
387  }
388  MPITIC; // allreduce gram
389  MPI_Allreduce(localWtW.memptr(), (*XtX).memptr(), this->k * this->k,
390  MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
391  temp = MPITOC; // allreduce gram
392  this->time_stats.communication_duration(temp);
393  this->time_stats.allreduce_duration(temp);
394  }
395  /*
396  * This is the main loop function
397  * Refer Algorithm 1 in Page 3 of
398  * the PPoPP HPC-NMF paper.
399  */
400  void computeNMF() {
401  PRINTROOT("computeNMF started");
402 
403 #ifdef MPI_VERBOSE
404  DISTPRINTINFO(PRINTMAT(this->A));
405 #endif
406  // error computation
407  if (this->is_compute_error()) {
408  prevH.zeros(size(this->H));
409  prevHtH.zeros(this->k, this->k);
410  WtAijH.zeros(this->k, this->k);
411  localWtAijH.zeros(this->k, this->k);
412  }
413 #ifdef __WITH__BARRIER__TIMING__
414  MPI_Barrier(MPI_COMM_WORLD);
415 #endif
416  for (int iter = 0; iter < this->num_iterations(); iter++) {
417  // saving current instance for error computation.
418  if (iter > 0 && this->is_compute_error()) {
419  this->prevH = this->H;
420  this->prevHtH = this->HtH;
421  }
422  MPITIC; // total_d W&H
423  // update H given WtW and WtA step 4 of the algorithm
424  {
425  // compute WtW
426  this->distInnerProduct(this->W, &this->WtW);
427  PRINTROOT(PRINTMATINFO(this->WtW));
428  this->applyReg(this->regH(), &this->WtW);
429 #ifdef MPI_VERBOSE
430  PRINTROOT(PRINTMAT(this->WtW));
431 #endif
432  // compute WtA
433  this->distWtA();
434  // PRINTROOT(PRINTMATINFO(this->WtAij));
435 #ifdef MPI_VERBOSE
436  DISTPRINTINFO(PRINTMAT(this->WtAij));
437 #endif
438  MPITIC; // nnls H
439  // ensure both Ht and H are consistent after the update
440  // some function find Ht and some H.
441  updateH();
442 #ifdef MPI_VERBOSE
443  DISTPRINTINFO("::it=" << iter << PRINTMAT(this->H));
444 #endif
445  double temp = MPITOC; // nnls H
446  this->time_stats.compute_duration(temp);
447  this->time_stats.nnls_duration(temp);
448  this->reportTime(temp, "NNLS::H::");
449  }
450  // Update W given HtH and AH step 3 of the algorithm.
451  {
452  // compute HtH
453  this->distInnerProduct(this->H, &this->HtH);
454  PRINTROOT("HtH::" << PRINTMATINFO(this->HtH));
455  this->applyReg(this->regW(), &this->HtH);
456 #ifdef MPI_VERBOSE
457  PRINTROOT(PRINTMAT(this->HtH));
458 #endif
459  // compute AH
460  this->distAH();
461  // PRINTROOT(PRINTMATINFO(this->AHtij));
462 #ifdef MPI_VERBOSE
463  DISTPRINTINFO(PRINTMAT(this->AHtij));
464 #endif
465  MPITIC; // nnls W
466  // Update W given HtH and AH step 3 of the algorithm.
467  // ensure W and Wt are consistent. As some algorithms
468  // determine W and some Wt.
469  updateW();
470 #ifdef MPI_VERBOSE
471  DISTPRINTINFO("::it=" << iter << PRINTMAT(this->W));
472 #endif
473  double temp = MPITOC; // nnls W
474  this->time_stats.compute_duration(temp);
475  this->time_stats.nnls_duration(temp);
476  this->reportTime(temp, "NNLS::W::");
477  }
478  this->time_stats.duration(MPITOC); // total_d W&H
479  if (iter > 0 && this->is_compute_error()) {
480 #ifdef BUILD_SPARSE
481  this->computeError(iter);
482 #else
483  this->computeError2(iter);
484 #endif
485 
486  PRINTROOT("it=" << iter << "::algo::" << this->m_algorithm << "::k::"
487  << this->k << "::err::" << sqrt(this->objective_err)
488  << "::relerr::"
489  << sqrt(this->objective_err / this->m_globalsqnormA));
490  }
491  PRINTROOT("completed it=" << iter
492  << "::taken::" << this->time_stats.duration());
493  } // end for loop
494  MPI_Barrier(MPI_COMM_WORLD);
495  this->reportTime(this->time_stats.duration(), "total_d");
496  this->reportTime(this->time_stats.communication_duration(), "total_comm");
497  this->reportTime(this->time_stats.compute_duration(), "total_comp");
498  this->reportTime(this->time_stats.allgather_duration(), "total_allgather");
499  this->reportTime(this->time_stats.allreduce_duration(), "total_allreduce");
500  this->reportTime(this->time_stats.reducescatter_duration(),
501  "total_reducescatter");
502  this->reportTime(this->time_stats.gram_duration(), "total_gram");
503  this->reportTime(this->time_stats.mm_duration(), "total_mm");
504  this->reportTime(this->time_stats.nnls_duration(), "total_nnls");
505  if (this->is_compute_error()) {
506  this->reportTime(this->time_stats.err_compute_duration(),
507  "total_err_compute");
508  this->reportTime(this->time_stats.err_compute_duration(),
509  "total_err_communication");
510  }
511  }

513  /*
514  * We assume this error function will be called in
515  * every iteration before updating the block to
516  * compute the error from previous iteration
517  */
518 
519  void computeError(const int it) {
520  // (init.norm_A)^2 - 2*trace(H*(A'*W))+trace((W'*W)*(H*H'))
521  // each process owns globalsqnormA will have (init.norm_A)^2
522  // each process owns WtAij is of size k*(globaln/p)
523  // each process owns H is of size (globaln/p)*k
524  // compute WtAij*H and do an MPI_ALL reduce to get the kxk matrix.
525  // every process local computation
526  MPITIC; // computeerror
527  this->localWtAijH = this->WtAij * this->prevH;
528 #ifdef MPI_VERBOSE
529  DISTPRINTINFO("::it=" << it << PRINTMAT(this->WtAij));
530  DISTPRINTINFO("::it=" << it << PRINTMAT(this->localWtAijH));
531  DISTPRINTINFO("::it=" << it << PRINTMAT(this->prevH));
532  PRINTROOT("::it=" << it << PRINTMAT(this->WtW));
533  PRINTROOT("::it=" << it << PRINTMAT(this->prevHtH));
534 #endif
535  double temp = MPITOC; // computererror
536  this->time_stats.err_compute_duration(temp);
537  MPITIC; // coommunication error
538  MPI_Allreduce(this->localWtAijH.memptr(), this->WtAijH.memptr(),
539  this->k * this->k, MPI_DOUBLE, MPI_SUM, MPI_COMM_WORLD);
540  temp = MPITOC; // communication error
541 #ifdef MPI_VERBOSE
